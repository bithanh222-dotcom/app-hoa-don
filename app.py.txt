import streamlit as st
from openai import OpenAI
import base64
import pandas as pd

client = OpenAI(api_key=st.secrets["OPENAI_API_KEY"])

st.title("Đọc hóa đơn")

files = st.file_uploader("Upload ảnh", accept_multiple_files=True)

if st.button("Xử lý"):
    all_data = []

    for file in files:
        img_base64 = base64.b64encode(file.read()).decode()

        res = client.chat.completions.create(
            model="gpt-4.1",
            messages=[
                {
                    "role": "user",
                    "content": [
                        {"type": "text", "text": "Trích SKU và số lượng, trả dạng: SKU | số lượng"},
                        {
                            "type": "image_url",
                            "image_url": {
                                "url": f"data:image/jpeg;base64,{img_base64}"
                            }
                        }
                    ]
                }
            ]
        )

        text = res.choices[0].message.content

        for line in text.split("\n"):
            if "|" in line:
                try:
                    sku, qty = line.split("|")
                    all_data.append([sku.strip(), float(qty.strip())])
                except:
                    pass

    df = pd.DataFrame(all_data, columns=["SKU", "SoLuong"])
    result = df.groupby("SKU").sum().reset_index()

    st.write(result)
